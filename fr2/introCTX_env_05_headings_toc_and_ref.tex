% This snippet defines a footer with page numbers. A name is assigned
% to it, and then, when section commands are configured, this name is
% assigned to the first page of part and chapter.
\definetext[ChapterFirstPage]
           [footer]
           [pagenumber]

\setuphead
  [part]
  [placehead=yes,
    bodypartlabel=part,
    alternative=middle,
%    conversion=I,
    style={\switchtobodyfont[36pt]\bf},
    color=purple,
    before={\ \godown[4cm]},
    header=high,
    footer=ChapterFirstPage,
  ]

\setuphead
  [title, chapter]
  [
    style={\switchtobodyfont[30pt]\bf},
    align=flushright,
    before={\blank[5*big, force]},
    after={\blank[2*big]},
    color=maincolor,
    sectionsegments=1:2,
    commandbefore={\blank[small]},
    header=high,
    footer=ChapterFirstPage,
  ]

\setuphead
  [section, subject]
  [
    page=yes,
    style={\bfc},
    before={\blank[2*big]},
    after=\blank,
    color=maincolor,
    sectionsegments=1:3,
  ]

\setuphead
  [subsection, subsubject]
  [
    style={\bfb},
    before=\blank,
    after=\blank,
    color=maincolor,
    sectionsegments=1:4,
  ]

  ]
