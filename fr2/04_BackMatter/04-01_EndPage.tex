\startcomponent 04-01_EndPage

\environment introCTX_env_00

%-------------------------------------------------------------------------------
                                                     % Front Cover with MetaPost
\startuseMPgraphic{BackCover}

StartPage ;

fill unitsquare xscaled PaperWidth yscaled PaperHeight withcolor \MPcolor{purple} ; 

draw textext.rt("\getvariable{document}{title}")
     xsized (.8PaperWidth)
     shifted (.1PaperWidth,.1PaperHeight)
     withcolor white;                    
                   

StopPage ;

\stopuseMPgraphic

%------------------------------------------------------------------------------

\defineoverlay[oBackCover][\useMPgraphic{BackCover}]

\setupbackgrounds [leftpage]  [background={background,oBackCover},state=repeat]
\setupbackgrounds [rightpage] [background={background},state=repeat]

\startstandardmakeup[page=left]
\stopstandardmakeup

%------------------------------------------------------------------------------

\stopcomponent
